svg.resize.js - Demo

Polygon, normal resizing

new SVG('polygon_normal').size('100%', '100%') .polygon('350,50 283,250 450,122 250,122 416,250') .select().select({deepSelect:true}) .resize();

Polygon, resizing wirg snapToGrid and snapToAngle

new SVG('polygon_snap').size('100%', '100%') .polygon('350,50 283,250 450,122 250,122 416,250') .select().select({deepSelect:true}) .resize({snapToGrid:20, snapToAngle:15});

Polygon, snaps on ctrl down

var poly = new SVG('polygon_ctrl').size('100%', '100%') .polygon('350,50 283,250 450,122 250,122 416,250') .select().select({deepSelect:true}) .resize(); window.addEventListener('keydown', function(e){ if(e.keyCode == 17){ poly.resize({'snapToGrid': 20, 'snapToAngle':15}); } }, false); window.addEventListener('keyup', function(e){ if(e.keyCode == 17){ poly.resize({'snapToGrid': 1,'snapToAngle':0.1}); } }, false);
